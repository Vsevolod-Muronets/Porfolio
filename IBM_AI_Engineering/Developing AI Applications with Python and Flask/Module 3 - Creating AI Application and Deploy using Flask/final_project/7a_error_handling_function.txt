def emotion_detector(text_to_analyse):
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    my_json_obj = { "raw_document": { "text": text_to_analyse } }
    resp = requests.post(url, json = my_json_obj, headers=headers)
    
    if resp.status_code == 400:
        emotions_scores = {
            "anger": None, "disgust": None, "fear": None,
            "joy": None, "sadness": None, "dominant_emotion": None
            }
    else:
        emotions_scores = json.loads(resp.text)['emotionPredictions'][0]['emotion']
        emotions_scores['dominant_emotion'] = max(emotions_scores, key=emotions_scores.get)
    
    return emotions_scores
